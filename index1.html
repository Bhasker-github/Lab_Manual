Comprehensive Java and Web Development Lab Manual - 12 Experiments

Experiment 1: Installation of Eclipse

  1. Visit the official Eclipse website: https://www.eclipse.org/downloads/
  2. Click on "Download Packages" and choose "Eclipse IDE for Java Developers"
  3. Select the appropriate version for your operating system (Windows, macOS, or Linux)
  4. Once downloaded, extract the ZIP file to your desired location
  5. For Windows and Linux: Run the eclipse.exe file
  6. For macOS: Drag Eclipse.app to your Applications folder and launch it from there
  7. When prompted, choose a workspace directory where your projects will be stored
  8. Eclipse will launch, and you're ready to start developing Java applications

Experiment 2: MySQL Installation

  1. Go to the MySQL download page: https://dev.mysql.com/downloads/mysql/
  2. Select your operating system and download the MySQL installer
  3. Run the installer and choose "Custom" setup type
  4. Select MySQL Server, Workbench, and Connector/J
  5. Click "Next" and "Execute" to download and install the selected products
  6. In the configuration step, choose "Development Computer" for the config type
  7. Set a root password when prompted
  8. Complete the installation and launch MySQL Workbench to verify the installation
  9. In Workbench, create a new connection using localhost, port 3306, and the root credentials

Experiment 7: Installing and Configuring Hibernate Tools in Eclipse

  1. Open Eclipse IDE
  2. Go to Help > Eclipse Marketplace
  3. Search for "JBoss Tools"
  4. Select "JBoss Tools" and click "Install"
  5. In the installation wizard, ensure "Hibernate Tools" is selected
  6. Accept the license agreements and click "Finish"
  7. Restart Eclipse when prompted
  8. To use Hibernate, right-click on your project > Properties
  9. Go to Project Facets and select "JPA" (Java Persistence API)
  10. Click "Further configuration available..." and select Hibernate as the platform

        SessionFactory factory = new Configuration()
                .configure("hibernate.cfg.xml")
                .addAnnotatedClass(YourEntity.class)
                .buildSessionFactory();

        try (Session session = factory.getCurrentSession()) {
            session.beginTransaction();
            // Perform database operations here
            session.getTransaction().commit();
            System.out.println("Connected to the database successfully!");
        } catch (Exception e) {
            System.out.println("Connection failed. Error: " + e.getMessage());

Experiment 10: Create & Insert Data in the Table (Server Side Programming)


import java.sql.*;

public class CreateInsertData {
    private static final String URL = "jdbc:mysql://localhost:3306/your_database";
    private static final String USER = "your_username";
    private static final String PASSWORD = "your_password";

    public static void main(String[] args) {
        try (Connection conn = DriverManager.getConnection(URL, USER, PASSWORD)) {
            // Create table
            String createTableSQL = "CREATE TABLE IF NOT EXISTS users (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(50), email VARCHAR(50))";
            try (Statement stmt = conn.createStatement()) {
                stmt.execute(createTableSQL);
                System.out.println("Table created successfully");
            }

            // Insert data
            String insertDataSQL = "INSERT INTO users (name, email) VALUES (?, ?)";
            try (PreparedStatement pstmt = conn.prepareStatement(insertDataSQL)) {
                pstmt.setString(1, "John Doe");
                pstmt.setString(2, "john@example.com");
                pstmt.executeUpdate();
                System.out.println("Data inserted successfully");
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}

            System.out.println("Email: " + rs.getString("email"));  
         }  
      }

 }

Experiment 12: Update and Delete Data in the Table (Server Side Programming)

  
import java.sql.*;  

public class UpdateDeleteData {  
 private static final String URL = "jdbc:mysql://localhost:3306/your_database";  
 private static final String USER = "your_username";  
 private static final String PASSWORD = "your_password";  

 public static void main(String[] args) {  
    try (Connection conn = DriverManager.getConnection(URL, USER, PASSWORD)) {  
      // Update data  
      String updateDataSQL = "UPDATE users SET name = ? WHERE id = ?";  
      try (PreparedStatement pstmt = conn.prepareStatement(updateDataSQL)) {  
         pstmt.setString(1, "Jane Doe");  
         pstmt.setInt(2, 1);  
         pstmt.executeUpdate();  
         System.out.println("Data updated successfully");  
      }  

      // Delete data  
      String deleteDataSQL = "DELETE FROM users WHERE id = ?";  
      try (PreparedStatement pstmt = conn.prepareStatement(deleteDataSQL)) {  
         pstmt.setInt(1, 1);  
         pstmt.executeUpdate();  
         System.out.println("Data deleted successfully");  
      }  
    } catch (SQLException e) {  
      e.printStackTrace();  
    }  
 }  
}
