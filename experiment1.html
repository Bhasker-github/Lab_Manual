Objective

Install Eclipse IDE for Java Development.

Steps

  1. Download Eclipse: Visit the Eclipse Downloads page and choose the "Eclipse IDE for Java Developers" package.
  2. Run the Installer: Execute the downloaded file and follow the prompts to install Eclipse.
  3. Launch Eclipse: Open Eclipse, set your workspace directory, and start exploring.

Code Example


    import java.util.Scanner;

    public class PrimeNumberChecker {
    
        public static void main(String[] args) {
            Scanner scanner = new Scanner(System.in);
            System.out.print("Enter a number: ");
            int number = scanner.nextInt();
    
            if (isPrime(number)) {
                System.out.println(number + " is a prime number.");
            } else {
                System.out.println(number + " is not a prime number.");
            }
    
            scanner.close();
        }
    
        public static boolean isPrime(int num) {
            if (num <= 1) {
                return false; // 0 and 1 are not prime numbers
            }
            for (int i = 2; i <= Math.sqrt(num); i++) {
                if (num % i == 0) {
                    return false; // Found a divisor, not prime
                }
            }
            return true; // No divisors found, it is prime
        }
    }
